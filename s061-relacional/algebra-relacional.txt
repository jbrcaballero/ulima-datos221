--CREACION DE TABLA
CREATE TABLE CLIENTE(
    CODIGO NUMBER(7) PRIMARY KEY,
    NOMBRE VARCHAR2(100),
    DNI VARCHAR2(8) UNIQUE,
    DIRECCION VARCHAR2(300),
    TELEFONO VARCHAR2(15),
    CIUDAD VARCHAR2(50),
    EDAD NUMBER(3)
);
